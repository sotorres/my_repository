# sklearn PCA
pca = PCA(n_components=3)
pca.fit(X)

D = pca.explained_variance_ #Matriz de autovalores; varianza explicada por cada componente

mu = pca.mean_
U = np.transpose(pca.components_)
perc = pca.singular_values_/np.sum(pca.singular_values_)
Y = pca.transform(X)
print(D,U,mu,perc,Y.shape)
